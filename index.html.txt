<script>

setTimeout(()=>{
    nextSection('intro');
},3000);

function nextSection(id){
    document.querySelectorAll('.section').forEach(sec=>sec.classList.remove('active'));
    document.getElementById(id).classList.add('active');
}

function decorateCake(){
    document.getElementById("cake").classList.add("decorated");
    document.getElementById("decorateBtn").style.display="none";
    document.getElementById("candleBtn").style.display="inline-block";
}

function addCandles(){
    let cake=document.getElementById("cake");
    for(let i=0;i<7;i++){
        let candle=document.createElement("div");
        candle.className="candle";
        candle.style.left=30 + i*25 + "px";

        let flame=document.createElement("div");
        flame.className="flame";
        candle.appendChild(flame);

        cake.appendChild(candle);
    }
    document.getElementById("candleBtn").style.display="none";
    document.getElementById("blowBtn").style.display="inline-block";
}

function blowCandles(){
    document.querySelectorAll(".flame").forEach(f=>f.remove());

    confetti({
        particleCount:150,
        spread:100,
        origin:{y:0.6}
    });

    setTimeout(()=>{
        nextSection('final');
    },2000);
}
